import java.util.Scanner;
public class Noofdigit{
  public static void main(String args[]){
    Scanner s=new Scanner(System.in);
    System.out.println("enter the number");
    int num=s.nextInt();
    int count=0;
    while(num!=0){
      num=num/10;
      count=count+1;
    }
    System.out.println(count);
    
  }
}
