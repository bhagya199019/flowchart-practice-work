import java.util.Scanner;
public class Even1ton{
  public static void main(String args[]){
    Scanner s=new Scanner(System.in);
    System.out.println("Enter the number");
    int num=s.nextInt();
    int i=1;
    while(i<=num)
      {
        if(i%2==0){
          System.out.println(i);
      }
        i++;
      }
    
  }
}
